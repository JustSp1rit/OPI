#include <iostream>
#include <iomanip>
#include "stdio.h"
#include "math.h" 

using namespace std;
int main()
{
double b, h, Y, x;

cout << "Введіть початкове значення x:\n";
cin >> x;
cout << "Введіть кінцеве значення x:\n";
cin >> b;
cout << "Введіть крок цикл:\n";
cin >> h;
while (x <= b)
{
Y = log(pow(x,3)) + 2*cos(x) - 2;
cout << "x = " << x << "\nY = " << Y << endl;

}

return 0;
}
